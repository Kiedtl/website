Add this script to somewhere in their $PATH and name is sudo:

#!/bin/bash

user=$(whoami)
attempts=0
trap 'printf "\nsudo: ${attempts} incorrect password attempts" && exit 1' EXIT
while :
do
    printf "[sudo] password for %s: " $user
    read -s pp
    printf "\nSorry, try again.\n"
    attempts=$(($attempts+1))
done
